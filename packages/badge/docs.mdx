## Examples

<div style={{display: 'grid', gridTemplateRows: '1fr', border: '1px solid black', borderRadius: '8px', padding: '16px' }}>
  {
    [
      'action',
      'neutral',
      'boundary',
      'consul',
      'nomad',
      'packer',
      'terraform',
      'vagrant',
      'vault',
      'waypoint',
    ].map((t, tIndex) => {
      return (
        <div key={tIndex} style={{display: 'grid', gridTemplateColumns: 'repeat(6, 1fr)'}}>
          {
            [
              'light',
              'faint',
              'strong',
              'strongFaint',
              'action',
              'actionFaint'
            ].map((p, pIndex) => {
              return ( 
                <div key={pIndex} style={{display: 'grid', gridTemplateRows: '1fr 1fr'}}>
                  {
                    [
                      'primary',
                      'secondary',
                    ].map((v, vIndex) => {
                      const isActionTheme = t === 'action';
                      const isNeutralTheme = t === 'neutral'
                      const isProduct = !isActionTheme && !isNeutralTheme
                      const pageToColorMap = {
                        'light': 'white',
                        'faint': '#EAEAEB',
                        'strong': 'var(--wpl-neutral-900)',
                        'strongFaint': 'var(--wpl-neutral-600)',
                        'action': isProduct ? `var(--${t})` : `#3571E9`,
                        'actionFaint': isProduct ? `var(--${t}-secondary)` : `#DFE8FB`,
                      }
                      const backgroundColor = pageToColorMap[p];

                      return (
                        <div key={vIndex} style={{ display: 'flex', justifyContent: 'center', alignItems: 'center', backgroundColor: backgroundColor, padding: '16px'}}>
                          <Badge variant={v} theme={t} page={p} text="Badge" />
                        </div>
                      )
                    })
                  }  
                </div>
              )
            })
          }
        </div>
      )
    })
  }
</div>
