## Example usage

```tsx
function Button() {
  const [isHovered, setIsHovered] = React.useState(false)
  return (
    <button
      onMouseOver={() => {
        setIsHovered(true)
      }}
      onMouseOut={() => {
        setIsHovered(false)
      }}
    >
      {children}
      <ExpandableArrow expanded={isHovered} />
    </button>
  )
}
```
